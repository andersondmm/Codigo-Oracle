
CREATE TABLE PACIENTE (
   IDPACIENTE                   INT NOT NULL,
   NOMBRE_PACIENTE              VARCHAR(50),
   APELLIDO_PACIENTE            VARCHAR(50),
   FECHA_NACIMIENTO             DATE,
   NUM_CEDULA         	        VARCHAR(50),
   NUM_HISTORIA_CLINICA         VARCHAR(50),
   TALLA	                VARCHAR(50),
   EMAIL			VARCHAR(50),
   
   CONSTRAINT PK_PACIENTE  PRIMARY KEY (IDPACIENTE)
)


CREATE TABLE CITAS(
   IDCITAS INT NOT NULL,
   ID_PACIENTE INT ,
   ID_DOCTOR  INT,
   FECHA_PROGRAMADA DATE,
   HORA VARCHAR(10),
   CONSTRAINT PK_CITAS  PRIMARY KEY  (IDCITAS));
   
   CREATE TABLE CONSULTAS (
   IDCONSULTAS          INT NOT NULL,
   ID_PACIENTE          INT,
   ID_DOCTOR            INT,
   MOTIVO				VARCHAR(50),
   CONSTRAINT PK_CONSULTAS PRIMARY KEY (IDCONSULTAS));
   
   CREATE TABLE DOCTOR (
   IDDOCTOR             INT NOT NULL,
   NOMBRE_DOCTOR        VARCHAR(50),
   APELLIDOS_DOCTOR     VARCHAR(50),
   TEL_CELULAR       	VARCHAR(50),
   CORREO               VARCHAR(50),
   ESPECIALIDAD     	VARCHAR(50),
   CONSTRAINT PK_DOCTOR PRIMARY KEY (IDDOCTOR)
)

 
   
   CREATE TABLE  ECOS (
   IDECOS               INT NOT NULL,
   ID_PACIENTE          INT,
   ID_DOCTOR            INT,
   DENSIDAD             VARCHAR(50),
   GLUCOSA              VARCHAR(50),
   NITRITOS             VARCHAR(50),
   EDAD_GESTACIONAL     VARCHAR(50),
   
   CONSTRAINT PK_ECOS PRIMARY KEY (IDECOS, ID_PACIENTE, ID_DOCTOR ));


   
   CREATE TABLE PARTO (
   IDPARTO              INT NOT NULL,
   ID_PACIENTE          INT,
   ID_DOCTOR            INT,
   FECHA_PARTO          DATE,
   TIPO_PARTO           VARCHAR(50),
   PESO       	        VARCHAR(50),
   TALLA		VARCHAR(50),
   SEXO                 VARCHAR(50),
   NOMBRE     	        VARCHAR(50),
   CONSTRAINT PK_PARTO PRIMARY KEY(IDPARTO,ID_PACIENTE));
